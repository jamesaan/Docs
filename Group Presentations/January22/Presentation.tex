\frame{
\frametitle{No long range order \emph{grace \`a} 
Mermin \& Wagner}

\begin{theorem}
Systems whose ground states break a continuous symmetry cannot display long 
range order in $d\leq 2$.
\end{theorem}
Why?
Goldstone's theorem: broken symmetries $\Longrightarrow$ low-energy excitations
\center{\textit{Long-wavelength spin waves wreck long-range order}}
}

%% 8 %%
\frame{ 
\frametitle{Phase transitions in 2D \emph{rely} on topological defects}
\begin{itemize}
 \item Long-wavelength fluctuations destroy order:
 \begin{itemize}
 \item XY: $\avg{e^{i\th}} = 0$
 \item Crystals: $\avg{e^{i\bv G \cdot \bv{u}(\br)}} = 0$, 
$\avg{e^{6i\varphi}} = 0$
\end{itemize}
 \item What distinguishes a disordered system from one that hasn't ``melted'' 
in two dimensions if there is no order?
 \begin{itemize}
 \item Stiffness (non-infinite susceptibility for XY, non-zero elastic 
constants for crystals)
 \item Power-law correlations (as opposed to exponential correlations)
 \end{itemize}
 \item They won't show up in the Hamiltonian unless you:
 \begin{itemize}
 \item understand the symmetries of the original, unapproximated Hamiltonian
 \item account for the topology of the order parameter space
 \end{itemize}
\end{itemize}
}
